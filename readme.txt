--------------------------------------------------
jQuery AutoFancyBOX Plugin
--------------------------------------------------

This plugin is a FancyBOX facility to allow plugin configuration from DOM itself.





--------------------------------------------------
JavaScript Example:
--------------------------------------------------

<script type="text/javascript">
$(document).ready(function(){
	$('.auto-fancybox').autoFancybox();
});
</script>


--------------------------------------------------
HTML Example:
--------------------------------------------------

<!-- IFRAME -->
<a href="http://www.consulenza-web.com/" title="Peg's Blog" 
	class="auto-fancybox" 
	data-type="iframe" 
	data-width="90%" 
	data-height="80%" 
	data-title="Blog di Marco Pegoraro"
>Marco's Blog</a>
